<color=#a44fff>from</color> random <color=#a44fff>import</color> shuffle
<color=#a44fff>from</color> DistancesGrid <color=#a44fff>import</color> DistancesGrid


<color=#a44fff>class</color> Circuit:
    <color=#808080>"""
    Class representing our road and it's score (how good it is). 
    Each city is given an unrepeatable index, therefore our road (here called genotype) 
    is represented as a list of those indexes.
    """</color>

    <color=#a44fff>def</color> __init__(<color=#66c2ff>self</color>, city_ids: <color=#ff6600>list</color>[<color=#ff6600>int</color>], distances_grid: DistancesGrid):
        <color=#66c2ff>self</color>.city_ids = city_ids
        <color=#66c2ff>self</color>.score = Circuit.get_score(city_ids, distances_grid)  

    <color=#ffa704>@staticmethod</color>
    <color=#a44fff>def</color> <color=#ffa704>get_score</color>(city_ids: <color=#ff6600>list</color>[<color=#ff6600>int</color>], distances_grid: DistancesGrid) -> <color=#ff6600>float</color>:
        <color=#808080>"""
        Method calculates the length of an entire road - for each id we take the following id,
        retreve distance between those cities and sum all the distances together.
        """</color>
        sum_distances = <color=#66c2ff>0.0</color>
        <color=#a44fff>for</color> city_id in <color=#ff6600>range</color>(<color=#ff6600>len</color>(city_ids)):  
            city_id_next = (city_id + <color=#66c2ff>1</color>) % <color=#ff6600>len</color>(city_ids)  
            sum_distances += distances_grid.get_distance(city_ids[city_id], city_ids[city_id_next])
        <color=#a44fff>return</color> sum_distances

    <color=#ffa704>@staticmethod</color>
    <color=#a44fff>def</color> <color=#ffa704>get_random</color>(distances_grid: DistancesGrid):
        random_genotype = <color=#ff6600>list</color>(<color=#ff6600>range</color>(distances_grid.number_of_cities))  
        shuffle(random_genotype)  
        <color=#a44fff>return</color> Circuit(random_genotype, distances_grid)

    <color=#ffa704>@staticmethod</color>
    <color=#a44fff>def</color> <color=#ffa704>get_best</color>(members: <color=#ff6600>list</color>):
        <color=#a44fff>return</color> <color=#ff6600>max</color>(members, key=<color=#a44fff>lambda</color> x: x.score)
